-- Duck pond analysis - Get our ducks in a row!
SELECT
  duck_type,
  COUNT(*) AS duck_count,
  AVG(quacks_per_minute) AS quack_rate,
  SUM(bread_consumed) AS total_bread
FROM pond_data
WHERE migration_season = 'Spring'
GROUP BY duck_type
ORDER BY quack_rate DESC;
